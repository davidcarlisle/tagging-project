\newcommand{\Essay}[3]{%
  \FlushRunningHeads
  \InitRunningHeads
  \ifthenelse{\equal{#1}{III}}{% [** TN: Hack to handle long running head]
    \ifthenelse{\boolean{ForPrinting}}{%
      \fancyhead[CE]{\RHeads{The Structure of the Atom and the}}%
      \fancyhead[CO]{\RHeads{Physical and Chemical Properties of the Elements}}%
    }{% Not ForPrinting
      \SetRunningHeads{The Structure of the Atom}%
    }%
  }{% Not Essay III
    \SetRunningHeads{#2}%
  }
  \BookMark{0}{Essay #1.}
  \renewcommand{\EssayNo}{#1}% For \Eq and \Tag
  \addtocontents{toc}{\protect\SectTitle{ESSAY #1}}
  \addtocontents{toc}{\protect\SubsectTitle[\protect\small]{\MakeUppercase{#2}}}
  \SectTitle[\Large]{ESSAY #1\footnotemark}
  \SubsectTitle{\MakeUppercase{#2}}
  \footnotetext{#3}%
}
